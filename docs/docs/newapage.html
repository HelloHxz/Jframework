1. 新增一个UI文件,命名为mynewpage.js
存放在 ～./Apps/Demo/pages/

define(["../logic/mynewpage"],function(pluginClass){
    return {
        pluginClass:pluginClass,
        style:{
            backgroundColor:"#f0f4f6"
        },
        root:["page_header","page_content","page_footer"],
        components:{
            page_header:{
                type:"view",
                root:["backIcon","header_title"],
                style:{
                    height:"44px",
                    borderBottom:"1px solid #e2e8ed",
                    backgroundColor:"#fff",
                    justifyContent:"center",
                    alignItems:"center"
                }
            },
            backIcon:{
              type:"icon",
              text:"返回",
              style:{
                position: "absolute",
                left: "0px",
                color: "rgb(0, 147, 255)",
                top: "7px",
                width: "50px",
                height: "30px",
              },
              textStyle: {
                fontSize: "14px"
              }
            },
            header_title:{
                type:"text",
                text:"新增页面示例",
                style:{
                    fontSize:14
                }
            },
            page_content:{
                type:"view",
                style:{
                    flex:1
                },

            },
            message_text:{
                type:"text",
                text:"这是一个新的页面",

                },
            },
            page_footer:{
                type:"view",
                style:{

        }
    };
});
2. 新增一个逻辑文件文件,命名为mynewpage.js
存放在 ～./Apps/Demo/logic/

define([],function(){
    function pageLogic(config){
        this.pageviewInstance = config.pageview;
        //获取到参数
        var urlparams = this.pageviewInstance.params;
    }
    pageLogic.prototype = {
        backIcon_click:function(sender,params){
            this.pageviewInstance.goBack();
        }
    };
    return pageLogic;
});
